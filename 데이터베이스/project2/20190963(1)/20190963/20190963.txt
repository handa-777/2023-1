CREATE TABLE customer (customer_ID VARCHAR(5) NOT NULL, cus_name VARCHAR(20) NOT NULL, address VARCHAR(50) NOT NULL, PRIMARY KEY (customer_ID));
CREATE TABLE package (package_ID VARCHAR(10) NOT NULL, package_type VARCHAR(20) NOT NULL, weight numeric(5,2) NOT NULL, timeliness DATE NOT NULL, content VARCHAR(20) NOT NULL, value NUMERIC(10,2) NULL, PRIMARY KEY (package_ID));
CREATE TABLE pack_order (package_ID VARCHAR(10) NOT NULL, order_date varchar(10) NOT NULL, order_year varchar(4) not null, order_month varchar(2) not null, price NUMERIC(10,2) NOT NULL, payment_type VARCHAR(20) NOT NULL, customer_ID VARCHAR(5) NOT NULL, PRIMARY KEY (package_ID), FOREIGN KEY (customer_ID) REFERENCES customer (customer_ID) on delete cascade, FOREIGN KEY (package_ID) REFERENCES package (package_ID) on delete cascade);
CREATE TABLE recipient (recipient_ID VARCHAR(5) NOT NULL, rec_name VARCHAR(20) NOT NULL, address VARCHAR(50) NOT NULL, PRIMARY KEY (recipient_ID));
CREATE TABLE receive (package_ID VARCHAR(10) NOT NULL, is_in_time NUMERIC(1,0) NOT NULL, recipient_ID VARCHAR(5) NOT NULL, PRIMARY KEY (package_ID), FOREIGN KEY (recipient_ID) REFERENCES recipient (recipient_ID) on delete cascade, FOREIGN KEY (package_ID) REFERENCES package (package_ID) on delete cascade);
CREATE TABLE vehicle (vehicle_ID VARCHAR(10) NOT NULL, vehicle_type VARCHAR(20) NOT NULL, PRIMARY KEY (vehicle_ID));
CREATE TABLE warehouse (warehouse_ID VARCHAR(10) NOT NULL, location VARCHAR(50) NOT NULL, PRIMARY KEY (warehouse_ID));
CREATE TABLE shipment (vehicle_ID VARCHAR(10) NOT NULL, warehouse_ID VARCHAR(10) NOT NULL, package_ID VARCHAR(10) NOT NULL, depart_time varchar(20) NOT NULL, PRIMARY KEY (vehicle_ID,warehouse_ID,package_ID), FOREIGN KEY (vehicle_ID) REFERENCES vehicle (vehicle_ID) on delete cascade, FOREIGN KEY (warehouse_ID) REFERENCES warehouse (warehouse_ID) on delete cascade, FOREIGN KEY (package_ID) REFERENCES package (package_ID) on delete cascade);
CREATE TABLE delivery (warehouse_ID VARCHAR(10) NOT NULL, vehicle_ID VARCHAR(10) NOT NULL, depart_time VARCHAR(20) NOT NULL, arrive_time VARCHAR(20) NULL, destination VARCHAR(50) NULL, PRIMARY KEY (warehouse_ID,vehicle_ID,depart_time), FOREIGN KEY (warehouse_ID) REFERENCES warehouse (warehouse_ID) on delete cascade, FOREIGN KEY (vehicle_ID) REFERENCES vehicle (vehicle_ID) on delete cascade);
